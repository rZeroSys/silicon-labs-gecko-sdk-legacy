777 // A library's status is an 8-bit value with information about it.
778 // The high bit indicates whether the library is present (1), or if it is a
779 // stub (0). The lower 7-bits can be used for codes specific to the library.
780 // This allows a library, like the security library, to specify what additional
781 // features are present.
782 // A value of 0xFF is reserved, it indicates an error in retrieving the
783 // library status.
784 #define EMBER_LIBRARY_PRESENT_MASK 0x80
785 #define EMBER_LIBRARY_IS_STUB 0x00
786 #define EMBER_LIBRARY_ERROR 0xFF
